// Express 서버 예시 (이건 로컬이나 VPS에서 실행) const express = require("express"); const axios = require("axios"); const app = express(); app.use(express.json()); app.post("/summarize", async (req, res) => { const userPrompt = req.body.prompt; const apiKey = "sk-..."; // 발급받은 GPT API Key try { const result = await axios.post("https://api.openai.com/v1/chat/completions", { model: "gpt-3.5-turbo", messages: [ { role: "system", content: "당신은 대화를 요약하는 조수입니다." }, { role: "user", content: userPrompt } ], temperature: 0.7, max_tokens: 100 }, { headers: { "Authorization": `Bearer ${apiKey}`, "Content-Type": "application/json" } }); const summary = result.data.choices[0].message.content.trim(); res.json({ summary }); } catch (error) { res.status(500).json({ error: "요약 실패" }); } }); app.listen(3000, () => { console.log("GPT 요약 서버 실행 중..."); });
