const express = require('express'); const fetch = require('node-fetch'); // 외부 요청을 처리하기 위해 node-fetch 사용 const app = express(); const port = 3000; // OpenAI API 키 const apiKey = 'sk-XXX'; // 발급받은 API 키 app.use(express.json()); app.post('/summarize', async (req, res) => { const { chatMessages } = req.body; const prompt = `다음은 최근 대화입니다. 핵심 주제를 요약하여 간단한 문장으로 알려주세요:\n\n${chatMessages}`; // OpenAI API 요청 const response = await fetch('https://api.openai.com/v1/chat/completions', { method: 'POST', headers: { 'Authorization': `Bearer ${apiKey}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ model: 'gpt-3.5-turbo', messages: [ { role: 'system', content: '당신은 대화를 요약하는 비서입니다.' }, { role: 'user', content: prompt } ], max_tokens: 100, temperature: 0.7 }) }); const data = await response.json(); const summary = data.choices[0].message.content.trim(); // 요약된 결과를 클라이언트로 반환 res.json({ summary }); }); app.listen(port, () => { console.log(`Server is running on port ${port}`); });
